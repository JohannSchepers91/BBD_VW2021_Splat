Hard Puzzle solution:

Repeat
Walk
if IsTileCurrent(junction) do 
(
if IsTileAhead(wall) do turn(right)
else do Turn(left)
)
if IsTileAhead(Splat) do turn(left)
until IsTileCurrent(Bank)

deposit(4) -- Banked Red splat

turn(around)
walk
turn(right)

repeat
walk
if IsTileAhead(splat) do turn(right)
if IsTileAhead(wall) do turn(left)
until IsTileCurrent(Bank)
deposit(2) --Banked Blue splat and made Purple

walk
turn(right)

repeat
walk
if IsTileCurrent(Junction) AND IsTileAhead(Wall) do Turn(right)
else ( if IsTileAhead(Wall) do turn(left)
if IsTileCurrent(Junction) AND IsTileAhead(Empty) do Turn(left) )
if IsTileAhead(Splat) do (turn(around) + Walk + turn(left))
until IsTileAhead(wall)

turn(around)
repeat
walk
if IsTileAhead(wall) do turn(left)
if IsTileAhead(wall) do turn(around)
until IsColor(Yellow) -- Yellow Splat

turn(right)
repeat
walk
if IsTileAhead(wall) do turn(left)
if IsTileAhead(wall) do turn(around)
until IsTileAhead(Gate)

turn(left) 
repeat
walk
if IsTileCurrent(Junction) do turn(right)
if IsTileAhead(wall) do turn(around)
until IsTileAhead(Splat)

turn(around)
